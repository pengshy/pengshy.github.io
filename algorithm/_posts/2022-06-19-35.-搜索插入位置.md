### 35. 搜索插入位置

#### 二分模板题

```c++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int len = nums.size();
        
        // 特判：当所有的数都小于target时，直接插入最后一个位置
        if (nums[len - 1] < target) return len;
        
        int l = 0, r = len - 1;
        while (l < r)
        {
            int mid = l + r >> 1;
            if (nums[mid] < target) l = mid + 1;
            else r = mid;
        }
        return l;
    }
};
```
